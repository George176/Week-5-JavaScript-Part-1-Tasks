/*Assigns userName to player name*/

const userName = "";

/*An if statement to decide which greeting to use*/

if (userName === "") { console.log('Hello!'); }

else { console.log(`Hello, ${userName}!`); }

/*Assigns the question to the variable userQuestion*/

const userQuestion = 'Will it rain?';

/*A message for the player confirming the question asked and player's name*/

console.log(`Good evening ${userName}! You asked "${userQuestion}"`);

/*The function to create a random number between 0 and 7*/

const randomNumber = Math.floor(Math.random() * 8);

/*Initialising the String variable eightBall*/

let eightBall = "";

/*A switch to assign the answer to the number generated by the random number generator*/

switch (randomNumber) {

  case 0:

    eightBall = "It is certain";

    break;

  case 1:

    eightBall = "It is decidedly so";

    break;

  case 2:

    eightBall = "Reply hazy try again";

    break;

  case 3:

    eightBall = "Cannot predict now";

    break;

  case 4:

    eightBall = "Do not count on it";

    break;

  case 5:

    eightBall = "My sources say no";

    break;

  case 6:

    eightBall = "Outlook not so good";

    break;

  case 7:

    eightBall = "Signs point to yes";

    break;

  default:

    eightBall = "An error has occured";

}

/*Printing out the assigned answer to the screen*/

console.log(`Answer: ${eightBall}`);
